const int emgPin = A0;
const int ledpin = 13;
int umbral = 350;

void setup() {
  Serial.begin(9600);
  pinMode(ledpin, OUTPUT);
}

void loop() {
  int emgValue = analogRead(emgPin); // Lectura del sensor

  if (emgValue > umbral) {
    digitalWrite(ledpin, HIGH); // Enciende el LED si se supera el umbral
  } else {
    digitalWrite(ledpin, LOW);  // Apaga el LED si no se supera
  }

  Serial.print("EMG Value: ");
  Serial.println(emgValue);

  delay(100);
}
